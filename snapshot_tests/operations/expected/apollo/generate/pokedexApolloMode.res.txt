  @@ocaml.warning("-32-30")
  module Raw = {
    type rec t_pokemon = {
      id: string,
      name: Js.Nullable.t<string>,
    }
    type rec t = {pokemon: Js.Nullable.t<t_pokemon>}
    type rec t_variables = unit
  }
  type rec t_pokemon = {
    id: string,
    name: option<string>,
  }
  type rec t = {pokemon: option<t_pokemon>}
  type rec t_variables = unit
  let query = "query   {
pokemon(name: "Pikachu")  {
id  
name  
}

}
"
  let parse = (
    (value): t => {
      pokemon: {
        let value = (value: Raw.t).pokemon
        switch Js.toOption(value) {
        | Some(value) =>
          Some(
            (
              {
                id: {
                  let value = (value: Raw.t_pokemon).id
                  value
                },
                name: {
                  let value = (value: Raw.t_pokemon).name
                  switch Js.toOption(value) {
                  | Some(value) => Some(value)
                  | None => None
                  }
                },
              }: t_pokemon
            ),
          )
        | None => None
        }
      },
    }: Raw.t => t
  )
  let serialize = (
    (value): Raw.t => {
      let pokemon = {
        let value = (value: t).pokemon
        switch value {
        | Some(value) =>
          Js.Nullable.return(
            (
              {
                let name = {
                  let value = (value: t_pokemon).name
                  switch value {
                  | Some(value) => Js.Nullable.return(value)
                  | None => Js.Nullable.null
                  }
                }
                and id = {
                  let value = (value: t_pokemon).id
                  value
                }
                {id, name}
              }: Raw.t_pokemon
            ),
          )
        | None => Js.Nullable.null
        }
      }
      {pokemon: pokemon}
    }: t => Raw.t
  )
  let serializeVariables = () => ()
  let makeVariables = () => ()
  let makeDefaultVariables = () => makeVariables()

  external toJson: Raw.t => Js.Json.t = "%identity"
  external variablesToJson: Raw.t_variables => Js.Json.t = "%identity"
